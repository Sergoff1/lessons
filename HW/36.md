#### Иерархии классов
##### Пример 1

У нас имеется иерархия Хуков, которые вносят дополнительную логику в обработку запроса. 
Мы можем на любой запрос навесить нужные хуки. Хуки делятся на две категории: те, что выполняются до обработки запроса и те, что выполняются после обработки запроса.
Эти категории обозначены классами `AbstractBeforeHook` и `AbstractAfterHook`. Однако ничего нового эти абстрактные классы не приносят в проект. 
Возможно слегка повышают наглядность. А значит от этих классов в иерархии можно безболезненно избавиться и унаследовать их потомков от базового класса `AbstractHook`.

```java
@Getter
@NoArgsConstructor
public class AbstractHook implements Hook {

    private HookConfig config;

    public AbstractHook(String hookName, Entity entity, EntityCommand command,
                        Map<String, String> initParams) {
        init(new HookConfigImpl(hookName, entity, command, initParams));
    }

    public void init(HookConfig config) {
        this.config = config;
    }
}

@NoArgsConstructor
public abstract class AbstractBeforeHook extends AbstractHook implements BeforeHook {

    protected AbstractBeforeHook(String hookName, Entity entity, EntityCommand command,
                                 Map<String, String> initParams) {
        super(hookName, entity, command, initParams);
    }

}

@NoArgsConstructor
public abstract class AbstractAfterHook extends AbstractHook implements AfterHook {

    protected AbstractAfterHook(String hookName, Entity entity, EntityCommand command,
                                Map<String, String> initParams) {
        super(hookName, entity, command, initParams);
    }

}

@Slf4j
public class UpdateSaInfoHook extends AbstractAfterHook {
    @Override
    public CommandResult doHook(ExecutionContext ctx, Map<String, String> params, CommandResult result) throws HookException {
        ctx.getSession().startTransaction();
        try {
            List<Long> childIds = CollectionsHelper.convertListToLong(
                    Arrays.asList(ParamHelper.getStringParam(params, "ID", "").split(",")));

            for (Long i : childIds) {
                ServiceAccountInfo.updateMacByServID(ctx.getSession(), i);
            }
            ctx.getSession().commit();

        } catch (Exception e) {
            log.error("{}", e);
            ctx.getSession().rollback();
            throw new HookException(e);
        } finally {
            ctx.getSession().close();
        }
        return result;
    }
}
```

##### Пример 2

Одной из базовых сущностей проекта является Команда. Команду можно вызвать с определёнными параметрами и контекстом исполнения.
По сути, каждый метод, который пользователь может вызвать через API, является командой. 
У нас сложилась целая иерархия команд:
Базовая `Command`, `DbCommand` для взаимодействия с базой данных, `WsCommand` для вызова биллингом. 
А конкретные команды наследуются от одного из этих классов. 
Так сложилось, что `WsCommand` не добавляет специфическую логику, а скорее является маркерным классом, который добавляет в иерархию наглядности.
Поэтому от `WsCommand` можно спокойно избавиться, а имеющиеся там статические методы перенести в утильный класс.


```java
public abstract class Command implements EntityCommand {

    protected Entity entity;
    protected Map<String, String> params;

    public Command() {
        this(null);
    }

    public Command(Entity entity) {
        this(entity, new LinkedHashMap<>());
    }

    public Command(Entity entity, Map<String, String> params) {
        this.entity = entity;
        this.params = new LinkedHashMap<>(params);
    }
    
    public abstract CommandResult invoke(ExecutionContext ctx, Map<String, String> params);
    
    //other logic
}

public abstract class DbCommand extends Command {
    //logic
}

public abstract class WsCommand extends DbCommand {
    public WsCommand(Entity entity) {
        super(entity);
    }

    public WsCommand(Entity entity, Map<String, String> params) {
        super(entity, params);
    }

    protected static boolean isNumberOfLength(int length, String number) throws Exception {
        return Pattern.compile("^\\d{" + length + "," + length + "}$").matcher(number).matches();
    }

    protected static boolean isNumberOfMaxLength(int length, String number) throws Exception {
        return Pattern.compile("^\\d{1," + length + "}$").matcher(number).matches();
    }

    public static CommandResult makeResultFromMap(Map<String, String> data) {
        return new CommandResult(Node.makeElement("rows", List.of(
                makeNodeFromMap("row", data).addChild(Node.makeAttribute("number", "0")))));
    }
    
    private static Node makeNodeFromMap(String name, Map<String, String> data) {
        Node foo = Node.makeElement(name);

        for (Map.Entry<String, String> e : data.entrySet()) {
            foo.addChild(Node.makeElement(e.getKey(), e.getValue()));
        }

        return foo;
    }
}

public class ServiceAccountByMacGet extends WsCommand {
    //logic
}
```

#### Interface Dispatch
Мы создали интерфейс Summable, который имеет один метод для сложения сущностей. Этот интерфейс могут 
реализовывать разные классы, которые не обязаны иметь отношение друг к другу.

```java
interface Summable<T> {
    T sum(T other);
}

class General implements Serializable {
    //...
}

class MyInteger extends General implements Summable<MyInteger> {
    private final Integer value;

    public MyInteger(Integer value) {
        this.value = value;
    }

    public Integer getValue() {
        return value;
    }

    @Override
    public MyInteger sum(MyInteger other) {
        return new MyInteger(this.value + other.getValue());
    }
}

class Vector<T extends General & Summable<T>> extends General implements Summable<Vector<T>> {

    private final List<T> elements;

    public Vector(List<T> elements) {
        this.elements = elements;
    }

    public T get(int index) {
        return elements.get(index);
    }

    public int getSize() {
        return elements.size();
    }

    public Vector<T> sum(Vector<T> other) {
        if (this.getSize() != other.getSize()) {
            return null;
        }

        List<T> result = IntStream.range(0, elements.size())
                .mapToObj(i -> elements.get(i).sum(other.get(i)))
                .toList();

        return new Vector<>(result);
    }
}
```
